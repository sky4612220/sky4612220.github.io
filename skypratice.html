# The following codes will clean up the text. 

jane1<-jane0 %>%
  slice(-c(1:5))%>% #머리말을 잘라내는 단계
  filter(!str_detect(text, "^\\s*$"))%>% #빈줄 제거 단계계
  mutate(text = str_replace_all(text, "_", " "))%>% #밑줄을 전부 공백으로 바꿈꿈
  mutate(
    text = str_replace_all(text, "!([^\\s])", "\\1"),          #느낌표 제거 공백 없게 조정)
    text = str_replace_all(text, "[[:punct:]&&[^.!]]", "")
  )%>% # 문장 끝 표시(.,!)만 남기고 나머지 기호 제거 
  distinct(text, .keep_all = TRUE)
  # 같은 문장이 여러번 있는 경우 하나만 남김
